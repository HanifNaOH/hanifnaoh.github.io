{{- $before := .Get "before" -}} {{- $after := .Get "after" -}} {{- if not (and $before $after) -}} {{- errorf "beforeAfter shortcode requires both 'before' and 'after' parameters" -}} {{- end -}} {{- $beforeAlt := .Get "beforeAlt" | default (.Get "beforeLabel") | default "Before" -}} {{- $afterAlt := .Get "afterAlt" | default (.Get "afterLabel") | default "After" -}} {{- $beforeLabel := .Get "beforeLabel" | default "Before" -}} {{- $afterLabel := .Get "afterLabel" | default "After" -}} {{- $caption := .Get "caption" -}} {{- $start := .Get "start" | default 50 -}} {{- $height := .Get "height" -}} {{- $fit := .Get "fit" | default "cover" -}} {{- $ariaLabel := .Get "ariaLabel" | default (printf "Drag to compare %s and %s" $beforeLabel $afterLabel) -}} {{- $uid := printf "before-after-%d" now.UnixNano -}} {{- $beforeSrc := $before -}} {{- if or (strings.HasPrefix $beforeSrc "http://") (strings.HasPrefix $beforeSrc "https://") -}} {{- $beforeSrc = $beforeSrc -}} {{- else if strings.HasPrefix $beforeSrc "/" -}} {{- $beforeSrc = relURL $beforeSrc -}} {{- else -}} {{- with $.Page.Resources.GetMatch $beforeSrc -}} {{- $beforeSrc = .RelPermalink -}} {{- else -}} {{- with resources.Get $beforeSrc -}} {{- $beforeSrc = .RelPermalink -}} {{- else -}} {{- $beforeSrc = relURL $beforeSrc -}} {{- end -}} {{- end -}} {{- end -}} {{- $afterSrc := $after -}} {{- if or (strings.HasPrefix $afterSrc "http://") (strings.HasPrefix $afterSrc "https://") -}} {{- $afterSrc = $afterSrc -}} {{- else if strings.HasPrefix $afterSrc "/" -}} {{- $afterSrc = relURL $afterSrc -}} {{- else -}} {{- with $.Page.Resources.GetMatch $afterSrc -}} {{- $afterSrc = .RelPermalink -}} {{- else -}} {{- with resources.Get $afterSrc -}} {{- $afterSrc = .RelPermalink -}} {{- else -}} {{- $afterSrc = relURL $afterSrc -}} {{- end -}} {{- end -}} {{- end -}}
